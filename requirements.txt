# Pydantic Settings - Settings management from environment variables
# Why: Loads .env files automatically, validates configuration
# Used by: app/config.py (BaseSettings class)
# Documentation: https://docs.pydantic.dev/latest/concepts/pydantic_settings/
pydantic-settings==2.7.0

# Python-dotenv - Load environment variables from .env files
# Why: Reads .env file and sets environment variables for the application
# Used by: Pydantic Settings (automatically), can be used directly if needed
# Documentation: https://pypi.org/project/python-dotenv/
python-dotenv==1.0.1

# -----------------------------------------------------------------------------
# DATABASE
# -----------------------------------------------------------------------------

# SQLAlchemy - SQL toolkit and Object-Relational Mapping (ORM) library
# Why: Interact with PostgreSQL database using Python objects instead of raw SQL
# Used by: app/db/models.py (define tables), app/db/session.py (connections)
# Documentation: https://docs.sqlalchemy.org/
# Note: Using 2.0+ syntax (modern, typed, async support)
sqlalchemy==2.0.23

# Psycopg2-binary - PostgreSQL database adapter for Python
# Why: SQLAlchemy needs this driver to connect to PostgreSQL
# Used by: SQLAlchemy engine (behind the scenes)
# Documentation: https://www.psycopg.org/
# Note: "binary" version includes pre-compiled C libraries (easier installation)
# Production note: Consider using psycopg2 (without -binary) for production
psycopg2-binary==2.9.9

# Alembic - Database migration tool for SQLAlchemy
# Why: Version control for database schema, safe schema changes, rollback support
# Used by: alembic/ directory, CLI commands (alembic upgrade/downgrade)
# Documentation: https://alembic.sqlalchemy.org/
# Note: Essential for production deployments, enables zero-downtime migrations
alembic==1.13.1
